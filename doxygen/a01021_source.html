42  * watchdog mode.

44 #include "contiki.h"
45 #include "reg.h"
46 #include "cpu.h"
47 #include "dev/smwdthrosc.h"
48 /*---------------------------------------------------------------------------*/
49 /** \brief Initialisation function for the WDT. Currently simply explicitly
50  * sets the WDT interval to max interval */
51 void
52 watchdog_init(void)
53 {
54  /* Max interval, don't enable yet */
55  REG(SMWDTHROSC_WDCTL) = 0;
56 }
57 /*---------------------------------------------------------------------------*/
58 /** \brief Starts the WDT in watchdog mode, maximum interval */
59 void
60 watchdog_start(void)
61 {
62  /* Max interval (32768), watchdog mode, Enable */
63  REG(SMWDTHROSC_WDCTL) = SMWDTHROSC_WDCTL_EN;
64 }
65 /*---------------------------------------------------------------------------*/
66 /** \brief Writes the WDT clear sequence. This function assumes that we are
67  * in watchdog mode and that interval bits (bits [1:0]) are 00 */
68 void
69 watchdog_periodic(void)
70 {
71  /* Safe to write to bits [3:0] since EN is 1 */
72  REG(SMWDTHROSC_WDCTL) = (SMWDTHROSC_WDCTL_CLR_3 | SMWDTHROSC_WDCTL_CLR_1);
73  REG(SMWDTHROSC_WDCTL) = (SMWDTHROSC_WDCTL_CLR_2 | SMWDTHROSC_WDCTL_CLR_0);
74 }
75 /*---------------------------------------------------------------------------*/
76 /** \brief In watchdog mode, the WDT can not be stopped. This function is
77  * defined here to satisfy API requirements.
78  */
79 void
80 watchdog_stop(void)
81 {
82  return;
83 }
84 /*---------------------------------------------------------------------------*/
85 /** \brief Keeps control until the WDT throws a reset signal */
86 void
87 watchdog_reboot(void)
88 {
89  INTERRUPTS_DISABLE();
90  while(1);
91 }
