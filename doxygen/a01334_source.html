109 {
110  uint8_t error;

112  PROCESS_BEGIN();

114  if((error = icmp6_new(NULL)) == 0) {

117 
118 #if UIP_CONF_IPV6

120 #else
121  if (1) {
122 #endif
123  LEDOn(LED2);
124  etimer_set(&et, CLOCK_SECOND/10);
125  } else {
126  LEDOff(LED2);

129  }
130  PROCESS_END();
131 }
